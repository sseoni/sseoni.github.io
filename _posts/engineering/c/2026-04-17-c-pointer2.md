// [Post 2 Code: 배열과 포인터, 연산자 우선순위]
#include <stdio.h>

int main() {
    int ary[3] = {10, 20, 30};

    // 1. 배열명은 포인터 상수
    printf("배열 시작 주소 (ary): %p\n", ary);
    printf("첫 번째 요소 주소 (&ary[0]): %p\n", &ary[0]);

    // 2. 포인터 연산과 연산자 우선순위
    int result1 = *ary + 3;    // *ary 값을 먼저 참조한 후 3을 더함
    int result2 = *ary+ 3;     // 위와 완전히 동일 (공백 무시)
    int result3 = *(ary + 1);  // 주소를 1칸(int 크기) 이동한 후 값을 참조

    printf("*ary + 3 의 결과: %d\n", result1); // 10 + 3 = 13
    printf("*ary+ 3 의 결과: %d\n", result2);  // 10 + 3 = 13
    printf("*(ary + 1) 의 결과: %d\n", result3); // ary[1]의 값 = 20

    return 0;
}

배열과 포인터의 관계, 그리고 연산자 우선순위에 대한 함정을 다룹니다.

배열명 = 시작 주소: C 언어에서 배열의 이름(예: ary)은 배열의 첫 번째 요소(&ary[0])를 가리키는 포인터 상수입니다.

배열과 포인터 표기법 호환: ary[1]과 *(ary + 1)은 컴파일러 입장에서 완벽히 동일한 코드입니다.

연산자 우선순위와 공백: *ary + 3;과 *ary+ 3;처럼 띄어쓰기가 어떻게 되어 있든 컴파일러는 공백을 무시합니다. 포인터 간접 참조 연산자(*)가 산술 연산자(+)보다 우선순위가 높기 때문에, 무조건 배열 첫 번째 값(*ary)을 먼저 읽어온 후 숫자 3을 더하게 됩니다. 주소값을 먼저 연산하고 싶다면 반드시 *(ary + 3)처럼 괄호를 사용해야 합니다.
